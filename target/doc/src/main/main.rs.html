use dnssector::*;

    let packet: Vec<u8> =
        vec![38, 44, 129, 160, 0, 1, 0, 2, 0, 0, 0, 1, 3, 99, 57, 120, 3, 111, 114, 71, 0, 0, 1,
             0, 1, 192, 12, 0, 1, 0, 1, 0, 0, 167, 29, 0, 4, 78, 194, 219, 1, 192, 12, 0, 46, 0,
             1, 0, 0, 167, 29, 0, 91, 0, 1, 13, 2, 0, 0, 168, 192, 89, 56, 71, 147, 89, 16, 186,
             147, 82, 60, 3, 99, 57, 120, 3, 111, 114, 103, 0, 153, 235, 139, 49, 43, 255, 159,
             252, 196, 189, 29, 77, 88, 132, 233, 31, 133, 88, 104, 42, 139, 12, 101, 158, 121,
             95, 105, 180, 59, 216, 202, 174, 113, 201, 121, 23, 4, 26, 241, 134, 233, 52, 104,
             120, 80, 237, 252, 215, 146, 44, 120, 229, 63, 16, 95, 19, 209, 103, 165, 196, 195,
             151, 222, 52, 0, 0, 41, 2, 0, 0, 0, 128, 0, 0, 0];
    let ds = DNSSector::new(packet).expect("cannot parse packet");
    let mut parsed_packet = ds.parse().expect("cannot run parser");
    {
        let it = parsed_packet.into_iter_question();
        let v = it.expect("next(1) failed");
        println!("{:?}", v.offset());
        assert!(v.next().is_none());
    }
    {
        let mut it = parsed_packet.into_iter_answer();
        while let Some(item) = it {
            println!("an item offset={:?} with ttl={} name={:?}",
                     item.offset(),
                     item.rr_ttl(),
                     item.name());
            it = item.next()
        }
    }
    {
        let mut it = parsed_packet.into_iter_nameservers();
        while let Some(item) = it {
            println!("ns item offset={:?}", item.offset());
            it = item.next()
        }
    }
    {
        let mut it = parsed_packet.into_iter_additional();
        while let Some(item) = it {
            println!("ar item offset={:?}", item.offset());
            it = item.next()
        }
    }
    {
        let mut it = parsed_packet.into_iter_edns();
        while let Some(item) = it {
            println!("edns item offset={:?}", item.offset());
            it = item.next()
        }
    }
